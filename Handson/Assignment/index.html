Q21. Armstrong Number?

        Ans. const CheckArm = (N) => {
            let sum = 0;
        
            let temp = N;
            while (temp > 0) {
                
                let remain = temp % 10;
            
                sum += remain * remain * remain;
            
                temp = parseInt(temp / 10); 
            }
            if (sum == N) {
                return "Yes";
            }
            else {
               return "No";
            }
        };

Q22. Unique Array?

        Ans. const UniqueElements = (N,arr) => 
        {
            for(let i= 0; i < N; i++){
            let count = 0;
            let k = arr[i]
            for(let j = 0; j < N; j++){
              
              if (arr[j] == k){
                
                count++;
                arr[j] = 0
              }
              
            }
            
            if (count >= 1 && k !== 0){
              
              console.log (k);
            }
          }
        };
